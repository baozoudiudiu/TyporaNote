// #ifdef MP-WEIXIN
				wx.initFaceDetect({
					success: () => {
						let context = uni.createCameraContext()
						let listen = context.onCameraFrame(function(res){
							wx.faceDetect({
								width: res.width,
								height: res.height,
								frameBuffer: res.data,
								complete: result => {
									uni.showModal({
										content: JSON.stringify(result)
									})
									listen.stop()
									wx.stopFaceDetect()
								}
							})
						})
						listen.start()
					}
				})
// #endif
